# Simulasi file lingkungan.py
class PesanLingkungan:
    def __init__(self, teks):
        self.teks = teks
    
    def olah_pesan(self):
        try:
            # Ubah ke huruf besar
            teks = self.teks.upper()
            # Split jadi list kata
            kata = teks.split()
            # Hitung vokal
            vokal = 0
            i = 0
            while i < len(teks):
                if teks[i].lower() in "aeiou":
                    vokal = vokal + 1
                i = i + 1
            # Ganti kata
            teks_ganti = teks.replace("PANTAI", "HUTAN")
            # Gabung kembali
            pesan_akhir = " ".join(teks_ganti.split())
            # Cek substring
            if "POHON" in pesan_akhir:
                print("Kata 'POHON' ada dalam pesan!")
            return pesan_akhir, vokal
        except:
            print("Terjadi error saat mengolah pesan!")
            return "", 0
